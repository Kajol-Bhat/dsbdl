Line Graph¶

In [ ]:
import matplotlib.pyplot as plt
import pandas as pd
data = pd.read_csv("airquality.csv") 
data["Ozone"].fillna(data['Ozone'].median(),inplace = True) 
h = data.iloc[1:16,1]
v = data.iloc[1:16,3] 
plt.plot(h,label = 'ozone',marker = 'o',linestyle = "dotted")
plt.plot(v,label = 'wind',marker = 'o',linestyle = "dashed") 
plt.ylim(0,60) 
plt.legend() 
plt.title("Ozone and Wind chart for 1st-15th May")
plt.ylabel("Value Range") 
plt.xlabel("Days")
plt.savefig("plot5.png")
plt.show()
